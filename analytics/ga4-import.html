⚙️ GA4 連携設定

← ダッシュボードに戻る
仕組み

Google Apps Script が毎日 GA4 Data API からデータを取得し、Supabase の ga4_history テーブルに自動書き込みします。
ダッシュボードは Supabase から読み取るだけなので、設定後は何もしなくても毎日更新されます。

GA4 → Apps Script（毎日自動）→ Supabase → ダッシュボード
Step 1 — Supabase にテーブルを作成
Supabase の SQL Editor で以下を実行してください。 
-- ga4_history テーブル作成（全指標対応版）
CREATE TABLE IF NOT EXISTS ga4_history (
  id bigserial PRIMARY KEY,
  date date NOT NULL,
  page text,
  country text,
  device_category text,
  source text,
  medium text,
  views integer DEFAULT 0,
  sessions integer DEFAULT 0,
  users integer DEFAULT 0,
  new_users integer DEFAULT 0,
  engaged_sessions integer DEFAULT 0,
  avg_engagement_sec float DEFAULT 0,
  bounce_rate float DEFAULT 0,
  synced_at timestamptz DEFAULT now(),
  UNIQUE(date, page, country, device_category, source, medium)
);
CREATE INDEX IF NOT EXISTS ga4_hist_date ON ga4_history (date);
CREATE INDEX IF NOT EXISTS ga4_hist_page ON ga4_history (page);
ALTER TABLE ga4_history ENABLE ROW LEVEL SECURITY;
CREATE POLICY IF NOT EXISTS "anon read" ON ga4_history FOR SELECT TO anon USING (true);
GRANT SELECT ON public.ga4_history TO anon;
GRANT SELECT ON public.ga4_history TO authenticated;
GRANT ALL ON public.ga4_history TO service_role;
GRANT ALL ON SEQUENCE public.ga4_history_id_seq TO service_role;
Step 2 — GA4 プロパティ ID を確認
Google Analytics を開く → 左下 管理（歯車アイコン） → プロパティの設定 → 「プロパティ ID」の数字（例: 123456789）をメモしてください。
Step 3 — Supabase の Service Role Key を取得
Supabase ダッシュボード → Project Settings → API → 「service_role」のキーをコピー。
⚠️ service_role キーは強い権限を持つため、Apps Script の Script Properties に保存します（ブラウザに保存しません）。

// ============================================================
// 28 Tools — GA4 → Supabase 自動同期スクリプト
// ============================================================

// ---- 初回のみ: 設定を保存 ----
// 以下の関数を一度だけ実行して設定を保存してください。
// その後は saveConfig() の実行は不要です。
function saveConfig() {
  PropertiesService.getScriptProperties().setProperties({
    GA4_PROPERTY_ID: '123456789',              // ← GA4 プロパティID（数字のみ）
    SUPABASE_URL:    'https://clyfrkwuajbauqxvrfyf.supabase.co',
    SUPABASE_KEY:    'eyJhbGciOiJIUzI1NiIsInR5...',  // ← service_role key
  });
  Logger.log('✅ 設定を保存しました');
}

// ---- 毎日自動実行: 昨日のデータを同期 ----
function syncYesterday() {
  const p = PropertiesService.getScriptProperties().getProperties();
  const yesterday = new Date();
  yesterday.setDate(yesterday.getDate() - 1);
  const dateStr = Utilities.formatDate(yesterday, 'Asia/Tokyo', 'yyyy-MM-dd');
  _syncDateRange(p, dateStr, dateStr);
  Logger.log('✅ 同期完了: ' + dateStr);
}

// ---- 初回のみ: 全期間の歴史データを一括取得 ----
function syncHistorical() {
  const p = PropertiesService.getScriptProperties().getProperties();
  _syncDateRange(p, '2020-01-01', 'yesterday');
  Logger.log('✅ 歴史データ同期完了');
}

// ---- 毎日のトリガーを設定（一度だけ実行） ----
function setupDailyTrigger() {
  // 既存トリガーを削除
  ScriptApp.getProjectTriggers().forEach(t => ScriptApp.deleteTrigger(t));
  // 毎日 午前2時（JST）= UTC 17時に実行
  ScriptApp.newTrigger('syncYesterday')
    .timeBased()
    .atHour(17)
    .nearMinute(0)
    .everyDays(1)
    .create();
  Logger.log('✅ 毎日トリガーを設定しました（JST 午前2時）');
}

// ============================================================
// 内部関数
// ============================================================
function _syncDateRange(p, startDate, endDate) {
  const PROPERTY_ID = p.GA4_PROPERTY_ID;
  const SUPABASE_URL = p.SUPABASE_URL;
  const SUPABASE_KEY = p.SUPABASE_KEY;

  let offset = 0;
  const LIMIT = 10000;

  while (true) {
    const report = AnalyticsData.Properties.runReport({
      dateRanges: [{ startDate, endDate }],
      dimensions: [
        { name: 'date' },
        { name: 'pagePath' },
        { name: 'country' },
        { name: 'deviceCategory' },
        { name: 'sessionSource' },
        { name: 'sessionMedium' },
      ],
      metrics: [
        { name: 'screenPageViews' },
        { name: 'sessions' },
        { name: 'totalUsers' },
        { name: 'newUsers' },
        { name: 'engagedSessions' },
        { name: 'averageSessionDuration' },
        { name: 'bounceRate' },
      ],
      limit: LIMIT,
      offset: offset,
    }, 'properties/' + PROPERTY_ID);

    const rows = (report.rows || []).map(row => {
      const d = row.dimensionValues[0].value;
      return {
        date: d.length === 8
          ? d.slice(0,4)+'-'+d.slice(4,6)+'-'+d.slice(6,8)
          : d,
        page:              row.dimensionValues[1].value || null,
        country:           row.dimensionValues[2].value || null,
        device_category:   row.dimensionValues[3].value || null,
        source:            row.dimensionValues[4].value || null,
        medium:            row.dimensionValues[5].value || null,
        views:             parseInt(row.metricValues[0].value) || 0,
        sessions:          parseInt(row.metricValues[1].value) || 0,
        users:             parseInt(row.metricValues[2].value) || 0,
        new_users:         parseInt(row.metricValues[3].value) || 0,
        engaged_sessions:  parseInt(row.metricValues[4].value) || 0,
        avg_engagement_sec:parseFloat(row.metricValues[5].value) || 0,
        bounce_rate:       parseFloat(row.metricValues[6].value) || 0,
      };
    });

    if (rows.length === 0) break;

    // Supabase に upsert（重複は上書き）
    const BATCH = 500;
    for (let i = 0; i < rows.length; i += BATCH) {
      const batch = rows.slice(i, i + BATCH);
      const res = UrlFetchApp.fetch(SUPABASE_URL + '/rest/v1/ga4_history', {
        method: 'post',
        contentType: 'application/json',
        headers: {
          'apikey': SUPABASE_KEY,
          'Authorization': 'Bearer ' + SUPABASE_KEY,
          'Prefer': 'resolution=merge-duplicates,return=minimal',
        },
        payload: JSON.stringify(batch),
        muteHttpExceptions: true,
      });
      if (res.getResponseCode() >= 400) {
        Logger.log('⚠️ Upsert エラー: ' + res.getContentText().slice(0, 200));
      }
    }

    Logger.log('  ' + rows.length + ' 行を処理 (offset=' + offset + ')');
    if (rows.length < LIMIT) break;
    offset += LIMIT;
    Utilities.sleep(500);
  }
}

実行順序
① コード内の GA4_PROPERTY_ID と SUPABASE_KEY を入力
② saveConfig を選んで実行（設定を保存）
③ syncHistorical を選んで実行（過去データを一括取得）
④ setupDailyTrigger を選んで実行（毎日自動化を設定）
→ 以後は何もしなくても毎朝2時に自動更新 ✅
